111, 135, WIDTH, HEIGHT, 5)    -- Create tween from myObj include easing    -- easing seems to control flow and action    -- of tween object while running        -- comment out all but the one fill(255)    text("tap screen to start or stop the tween",WIDTH/2,900)    text("triple tap to restart",WIDTH/2,850) --   ellipse(100,100,10)    -- visual starting point --   ellipse(700,900,10)    -- visual ending point    scale(1)    -- The next three lines are used for a sprite with rotation--    translate(myObj["start"].x, myObj["start"].y)--    rotate(myObj["rot"])--    sprite("Tyrian Remastered:Plane Boss", 0,0)    -- The next line draws a circle    ellipse(myObj["start"].x,myObj["start"].y, 40)    -- draw tween subject  --  ellipse(point.x, point.y, 40)    popMatrix()end--# ObjSetupObjSetup = class()function ObjSetup:init(strtx, strty, rot, endx, endy, ttime)    -- Create object as a table    Obj = {}        -- Set start x and y    Obj["start"] = {x = strtx, y = strty}    -- Set rototation    Obj["rot"] = rot    -- Set ending x and y    Obj["dest"] = {x = endx, y = endy}    -- Set run time    Obj["time"] = ttime        -- Return object table    return Obj    end
